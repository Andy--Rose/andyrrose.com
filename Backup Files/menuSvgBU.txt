		<script type="text/javascript">
			var currName = "Main";
			function setStart() {
				setActiveLink("Main");
			}
			function navOver(event) {
	 			var nav = event.target;
				nav.setAttribute("stroke", "#E6E62E");
				nav.setAttribute("fill", "#00FFFF");
	    		}
			function navOut(event){
				var nav = event.target;
				nav.setAttribute("stroke", "#3D001F");
				nav.setAttribute("fill", "#B53375");	
			}
			function textOver(event){
	 			var nav = event.target;
				var par = nav.parentNode;
				var bar = par.childNodes.item(1);
				nav.setAttribute("stroke", "#E6E62E");
				nav.setAttribute("fill", "#00FFFF");
				bar.setAttribute("stroke", "#E6E62E");
				bar.setAttribute("fill", "#00FFFF");						
			}
			function textOut(event){
	 			var nav = event.target;
				var par = nav.parentNode;
				var bar = par.childNodes.item(1);
				nav.setAttribute("stroke", "none");
				nav.setAttribute("fill", "white");
				bar.setAttribute("stroke", "#3D001F");
				bar.setAttribute("fill", "#B53375");
			}
			function resetLink(name){
				var box = document.getElementById(name + "Box");
				box.setAttribute("fill", "#B53375");
				box.setAttribute("stroke", "#3D001F");
				box.setAttribute("onmouseover", "navOver(evt)");
				box.setAttribute("onmouseout", "navOut(evt)");
				var text = document.getElementById(name + "Text");
				text.setAttribute("fill", "white");
				text.setAttribute("stroke", "none");
				text.setAttribute("onmouseover", "textOver(evt)");
				text.setAttribute("onmouseout", "textOut(evt)");
			}
			function setActiveLink(name){
				var nextBar = document.getElementById(name + "Box");
				nextBar.setAttribute("stroke", "#E6E62E");
				nextBar.setAttribute("fill", "#00FFFF");
				nextBar.setAttribute("onmouseover", "doNothing(evt)");
				nextBar.setAttribute("onmouseout", "doNothing(evt)");	
				var nextText = document.getElementById(name + "Text");
				nextText.setAttribute("stroke", "black");
				nextText.setAttribute("fill", "white");
				nextText.setAttribute("onmouseover", "doNothing(evt)");
				nextText.setAttribute("onmouseout", "doNothing(evt)");
				currName = name;
			}
			function setNewPage(event){
				var next = event.target;
				var parent = next.parentNode;
				var nextName = parent.getAttributeNS(null, 'id');
				// if loop for clicking the same link
				if (currName != nextName)
				{
					// Reset the previous page link to be normal
					resetLink(currName);
					
					// Set the new link to be active
					setActiveLink(nextName);
				}
			}
			function doNothing(event){ var busy = 0;}
		</script>
